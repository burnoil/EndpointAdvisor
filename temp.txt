// 1) Copy files into place
waithidden mkdir "C:\Program Files\LLNotify"
copy __Download\LLNOTIFY.ps1      "C:\Program Files\LLNotify\LLNOTIFY.ps1"
copy __Download\LL_LOGO.ico       "C:\Program Files\LLNotify\LL_LOGO.ico"
copy __Download\LL_LOGO_MSG.ico   "C:\Program Files\LLNotify\LL_LOGO_MSG.ico"

// 2) Create shortcut in current user's Startup folder (as a single-line PowerShell command)
override runas=currentuser
run powershell.exe -NoProfile -ExecutionPolicy Bypass -Command "$s=(New-Object -ComObject WScript.Shell).CreateShortcut([Environment]::GetFolderPath('Startup') + '\LLNotify.lnk');$s.TargetPath='%SystemRoot%\System32\WindowsPowerShell\v1.0\powershell.exe';$s.Arguments='-WindowStyle Hidden -NoProfile -ExecutionPolicy Bypass -File \"C:\Program Files\LLNotify\LLNOTIFY.ps1\"';$s.IconLocation='C:\Program Files\LLNotify\LL_LOGO.ico';$s.Save()"

// 3) Launch LLNOTIFY immediately under the current user
override runas=currentuser
run "%SystemRoot%\System32\WindowsPowerShell\v1.0\powershell.exe" -NoProfile -WindowStyle Hidden -ExecutionPolicy Bypass -File "C:\Program Files\LLNotify\LLNOTIFY.ps1"
